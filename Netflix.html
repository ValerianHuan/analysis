from collections import Counter
from wordcloud import WordCloud
df_clean = df.dropna(subset=['cast']).copy()
df_clean['cast'] = df_clean['cast'].str.split(',').apply(lambda x: [i.strip() for i in x])
all_actors = list(chain.from_iterable(df_clean['cast']))
# 3. 5
actor_counts = Counter(all_actors)
filtered_actors = {k: v for k, v in actor_counts.items() if v >= 5}  # {:  for ,  in  if }

#
wordcloud_actors = WordCloud(
    width=800,
    height=500,
    background_color='black',
    max_words=50,
    colormap='rainbow'
).generate_from_frequencies(filtered_actors)
plt.figure(figsize=(10, 6))
plt.imshow(wordcloud_actors, interpolation='bilinear')
plt.axis('off')
plt.title('', fontsize=14)
plt.show()
